async def decompose_topic(
    self,
    query,
    knowledge_gap,
    research_loop_count,
    uploaded_knowledge=None,
    existing_tasks=None,
):
    """分解研究主题"""
    from src.prompts import query_writer_instructions
    from llm_clients import get_async_llm_client, CURRENT_DATE, CURRENT_YEAR, ONE_YEAR_AGO
    
    # 准备上下文
    AUGMENT_KNOWLEDGE_CONTEXT = ""
    if uploaded_knowledge:
        AUGMENT_KNOWLEDGE_CONTEXT = f"""
        USER-PROVIDED EXTERNAL KNOWLEDGE AVAILABLE:
        {uploaded_knowledge[:500]}...
        """
    
    # 数据库上下文
    database_context = ""
    if hasattr(self, "database_info") and self.database_info:
        database_list = []
        for db in self.database_info:
            db_info = f"- {db.get('filename')} ({db.get('file_type')}) with {len(db.get('tables', []))} tables"
            database_list.append(db_info)
        database_context = f"""
        DATABASE AVAILABLE:
        {chr(10).join(database_list)}
        
        TOOL SELECTION:
        - Questions about uploaded data → use "text2sql"
        - Questions about external info → use search tools
        """
    
    # Steering 上下文
    steering_context = ""
    if hasattr(self, "state") and self.state and hasattr(self.state, "steering_todo"):
        steering_context = self.state.get_steering_plan()
    
    # 格式化 Prompt
    prompt = query_writer_instructions.format(
        current_date=CURRENT_DATE,
        current_year=CURRENT_YEAR,
        one_year_ago=ONE_YEAR_AGO,
        AUGMENT_KNOWLEDGE_CONTEXT=AUGMENT_KNOWLEDGE_CONTEXT,
        DATABASE_CONTEXT=database_context,
        research_topic=query,
        research_context=research_context,
        steering_context=steering_context,
    )
    
    # 调用 LLM
    llm_client = get_async_llm_client(provider, model)
    response = await llm_client.ainvoke(prompt)
    
    # 解析 JSON 响应
    decomposition = json.loads(response)
    
    return {
        "topic_complexity": decomposition.get("topic_complexity"),
        "subtopics": decomposition.get("subtopics", []),
        "main_query": decomposition.get("main_query"),
    }
```

async def reflect_on_report(state: SummaryState):
    """反思报告完整性"""
    from src.prompts import reflection_instructions
    from llm_clients import get_llm_client, CURRENT_DATE, CURRENT_YEAR, ONE_YEAR_AGO
    
    # 准备任务队列信息
    pending_tasks = ""
    completed_tasks = ""
    steering_messages = ""
    
    if hasattr(state, "steering_todo") and state.steering_todo:
        # 获取待处理任务
        pending = state.steering_todo.get_pending_tasks()
        pending_tasks = "\n".join([
            f"- [{t.task_id}] {t.description}" 
            for t in pending
        ])
        
        # 获取已完成任务
        completed = state.steering_todo.get_completed_tasks()
        completed_tasks = "\n".join([
            f"- [{t.task_id}] {t.description}" 
            for t in completed
        ])
        
        # 获取用户引导消息
        messages = state.steering_todo.pending_messages
        steering_messages = "\n".join([
            f"[{i}] {msg.get('content', '')}" 
            for i, msg in enumerate(messages)
        ])
    
    # 格式化 Prompt
    prompt = reflection_instructions.format(
        current_date=CURRENT_DATE,
        current_year=CURRENT_YEAR,
        one_year_ago=ONE_YEAR_AGO,
        research_topic=state.research_topic,
        current_summary=state.running_summary,
        pending_tasks=pending_tasks or "No pending tasks.",
        completed_tasks=completed_tasks or "No completed tasks yet.",
        steering_messages=steering_messages or "No steering messages.",
    )
    
    # 调用 LLM
    configurable = get_configurable(state.config)
    llm_client = get_llm_client(
        configurable.llm_provider,
        configurable.llm_model
    )
    
    response = await llm_client.ainvoke(prompt)
    
    # 解析 JSON 响应
    reflection_result = json.loads(response)
    
    # 更新任务队列
    if hasattr(state, "steering_todo") and state.steering_todo:
        # 标记完成的任务
        for task_id in reflection_result.get("mark_completed", []):
            state.steering_todo.mark_task_completed(task_id)
        
        # 取消任务
        for task_id in reflection_result.get("cancel_tasks", []):
            state.steering_todo.cancel_task(task_id)
        
        # 添加新任务
        for task_data in reflection_result.get("add_tasks", []):
            state.steering_todo.add_task(
                description=task_data["description"],
                rationale=task_data.get("rationale", ""),
                source=task_data.get("source", "knowledge_gap"),
            )
    
    return {
        "knowledge_gap": reflection_result.get("knowledge_gap", ""),
        "search_query": reflection_result.get("next_query", ""),
        "research_complete": reflection_result.get("research_complete", False),
    }
```

async def execute_research(self, state, callbacks=None, database_info=None):
    """执行研究任务"""
    # 存储状态和数据库信息
    self.state = state
    self.database_info = database_info
    
    # 创建初始研究计划（如果启用了 Steering）
    if hasattr(state, "steering_todo") and state.steering_todo:
        await self._create_initial_research_plan(state)
    
    # 获取研究计划
    if not state.research_plan:
        # 首次分解主题
        decomposition = await self.decompose_topic(
            query=state.research_topic,
            knowledge_gap=state.knowledge_gap,
            research_loop_count=state.research_loop_count,
            uploaded_knowledge=state.uploaded_knowledge,
        )
        state.research_plan = decomposition
    
    # 创建工具执行器
    from src.graph import ToolRegistry, ToolExecutor
    tool_registry = ToolRegistry(self.config)
    tool_executor = ToolExecutor(tool_registry)
    
    # 创建搜索智能体
    search_agent = SearchAgent(self.config, database_info)
    
    # 执行子任务
    subtasks = state.research_plan.get("subtopics", [])
    results = []
    
    # 并行执行子任务
    tasks = []
    for subtask in subtasks:
        task = search_agent.execute(subtask, tool_executor)
        tasks.append(task)
    
    # 等待所有任务完成
    search_results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # 处理结果
    for idx, result in enumerate(search_results):
        if isinstance(result, Exception):
            self.logger.error(f"Subtask {idx} failed: {result}")
            continue
        
        results.append(result)
        # 更新状态
        if result.get("sources"):
            state.sources_gathered.extend(result["sources"])
        if result.get("content"):
            state.web_research_results.append({
                "title": result.get("title", ""),
                "url": result.get("url", ""),
                "content": result.get("content", ""),
            })
    
    return state
```

async def execute(self, subtask, tool_executor=None):
    """执行搜索子任务"""
    # 从子任务中提取参数
    query = subtask.get("query")
    tool_name = subtask.get("suggested_tool", subtask.get("tool", "general_search"))
    subtask_name = subtask.get("name", "Unnamed subtask")
    
    self.logger.info(f"[SearchAgent] Executing: {subtask_name} with {tool_name}")
    
    # 根据工具类型执行
    if tool_name == "text2sql":
        # 数据库查询
        db_id = None
        if self.database_info and len(self.database_info) > 0:
            db_id = self.database_info[0].get("id")
        
        result = await self.text2sql_search(query, db_id)
    elif tool_name == "academic_search":
        result = await tool_executor.execute_tool(
            "academic_search",
            {"query": query, "top_k": 5}
        )
    elif tool_name == "github_search":
        result = await tool_executor.execute_tool(
            "github_search",
            {"query": query, "top_k": 5}
        )
    elif tool_name == "linkedin_search":
        result = await tool_executor.execute_tool(
            "linkedin_search",
            {"query": query, "top_k": 5}
        )
    else:
        # 默认使用通用搜索
        result = await tool_executor.execute_tool(
            "general_search",
            {"query": query, "top_k": 5}
        )
    
    return {
        "subtask_name": subtask_name,
        "query": query,
        "tool": tool_name,
        "result": result,
    }
```

async def async_multi_agents_network(state: SummaryState, callbacks=None):
    """多智能体网络入口点"""
    logger.info("[async_multi_agents_network] Starting research")
    
    # 获取配置
    configurable = get_configurable(state.config)
    
    # 获取数据库信息
    database_info = get_database_info()
    if database_info:
        logger.info(f"[async_multi_agents_network] Database info available: {len(database_info)} databases")
    
    # 创建主控智能体
    master_agent = MasterResearchAgent(config=state.config)
    
    # 执行研究
    try:
        await master_agent.execute_research(
            state=state,
            callbacks=callbacks,
            database_info=database_info,
        )
        
        # 更新研究循环计数
        state.research_loop_count += 1
        
        logger.info(f"[async_multi_agents_network] Research loop {state.research_loop_count} completed")
        
    except Exception as e:
        logger.error(f"[async_multi_agents_network] Error: {e}")
        traceback.print_exc()
    
    return state
```

def route_research(state: SummaryState, config: RunnableConfig):
    """决定研究是否继续"""
    configurable = Configuration.from_runnable_config(config)
    
    # 获取最大循环数
    max_loops = get_max_loops(
        configurable,
        state.extra_effort,
        state.minimum_effort,
        state.benchmark_mode,
        state.qa_mode,
    )
    
    # 检查是否达到最大循环数
    if state.research_loop_count >= max_loops:
        logger.info(f"Max loops reached ({max_loops}), finalizing report")
        return "finalize_report"
    
    # 检查 LLM 是否标记为完成
    if state.research_complete:
        logger.info("Research marked as complete by LLM, finalizing report")
        return "finalize_report"
    
    # 第一轮强制继续
    if state.research_loop_count == 1:
        logger.info("First iteration - forcing research to continue")
        return "multi_agents_network"
    
    # 如果没有后续查询，完成研究
    if not state.search_query or not state.search_query.strip():
        logger.info("No follow-up query, finalizing report")
        return "finalize_report"
    
    # 继续研究
    logger.info("Continuing research with follow-up query")
    return "multi_agents_network"
```

def create_graph():
    """创建 LangGraph 状态图"""
    # 创建图构建器
    builder = StateGraph(
        SummaryState,
        input=SummaryStateInput,
        output=SummaryStateOutput,
        config_schema=Configuration,
    )
    
    # 添加所有节点
    builder.add_node("multi_agents_network", async_multi_agents_network)
    builder.add_node("generate_report", generate_report)
    builder.add_node("reflect_on_report", reflect_on_report)
    builder.add_node("finalize_report", finalize_report)
    builder.add_node("generate_answer", generate_answer)
    builder.add_node("reflect_answer", reflect_answer)
    builder.add_node("finalize_answer", finalize_answer)
    builder.add_node("validate_context_sufficiency", validate_context_sufficiency)
    builder.add_node("refine_query", refine_query)
    
    # 添加边
    builder.add_edge(START, "multi_agents_network")
    
    # 多智能体网络后的条件路由
    def route_after_multi_agents_decision(state):
        if state.qa_mode or state.benchmark_mode:
            return "validate_context_sufficiency"
        else:
            return "generate_report"
    
    builder.add_conditional_edges(
        "multi_agents_network",
        route_after_multi_agents_decision,
        {
            "validate_context_sufficiency": "validate_context_sufficiency",
            "generate_report": "generate_report",
        },
    )
    
    # Benchmark 路径路由
    builder.add_conditional_edges(
        "validate_context_sufficiency",
        lambda state: "refine_query" if state.needs_refinement else "generate_answer",
        {"refine_query": "refine_query", "generate_answer": "generate_answer"},
    )
    builder.add_edge("refine_query", "multi_agents_network")
    
    # 常规报告流程
    builder.add_edge("generate_report", "reflect_on_report")
    builder.add_conditional_edges(
        "reflect_on_report",
        route_research,
        {
            "multi_agents_network": "multi_agents_network",
            "finalize_report": "finalize_report",
        },
    )
    builder.add_edge("finalize_report", END)
    
    # Benchmark 流程
    builder.add_edge("generate_answer", "reflect_answer")
    builder.add_conditional_edges(
        "reflect_answer",
        lambda state: route_after_reflect_answer(state, {}),
        {
            "multi_agents_network": "multi_agents_network",
            "finalize_answer": "finalize_answer",
        },
    )
    builder.add_edge("finalize_answer", END)
    
    # 编译并返回图
    return builder.compile()
```

## 5. Text2SQL 工具实现

**文件**: `src/tools/text2sql_tool.py`

```python
class Text2SQLTool:
    """Text2SQL 工具实现"""
    
    def query_database(self, db_id: str, natural_language_query: str) -> Dict[str, Any]:
        """查询数据库"""
        if db_id not in self.databases:
            raise ValueError(f"Database {db_id} not found")
        
        db_info = self.databases[db_id]
        schema = db_info["metadata"]
        
        # 使用 LLM 生成 SQL
        sql_query = self._generate_sql(natural_language_query, schema)
        
        # 执行 SQL 查询
        results = self._execute_sql(db_id, sql_query)
        
        return {
            "query": natural_language_query,
            "sql": sql_query,
            "results": results,
            "database": db_info["filename"],
        }
    
    def _generate_sql(self, natural_language_query: str, schema: Dict[str, Any]) -> str:
        """使用 LLM 生成 SQL"""
        # 格式化模式信息
        schema_description = self._format_schema_for_llm(schema)
        
        prompt = f"""
You are a SQL expert. Convert the following natural language query to SQL.

Database Schema:
{schema_description}

Natural Language Query: {natural_language_query}

CRITICAL INSTRUCTIONS:
1. Generate ANALYTICAL queries with insights
2. Use aggregations (COUNT, AVG, SUM, MAX, MIN) when appropriate
3. ALWAYS add LIMIT 20
4. Return ONLY the SQL query, no explanations

SQL Query:
"""
        
        # 调用 LLM
        llm_client = self._get_llm_client()
        response = llm_client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            max_tokens=500,
            temperature=0.1,
        )
        
        sql_query = response.choices[0].message.content.strip()
        
        # 清理 SQL（移除 markdown 格式）
        if sql_query.startswith("```sql"):
            sql_query = sql_query[6:]
        if sql_query.endswith("```"):
            sql_query = sql_query[:-3]
        
        return sql_query.strip()
    
    def _execute_sql(self, db_id: str, sql_query: str) -> Dict[str, Any]:
        """执行 SQL 查询"""
        db_info = self.databases[db_id]
        
        # 确定数据库文件路径
        if db_info["file_type"] == "sqlite":
            db_path = db_info["file_path"]
        elif db_info["file_type"] in ["csv", "json"]:
            db_path = db_info["metadata"]["sqlite_path"]
        else:
            raise ValueError(f"Unsupported database type: {db_info['file_type']}")
        
        # 执行查询
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()
        cursor.execute(sql_query)
        
        if sql_query.strip().upper().startswith("SELECT"):
            results = cursor.fetchall()
            columns = [description[0] for description in cursor.description]
            
            formatted_results = []
            for row in results:
                formatted_results.append(dict(zip(columns, row)))
            
            return {
                "type": "select",
                "columns": columns,
                "rows": formatted_results,
                "row_count": len(formatted_results),
            }
        else:
            conn.commit()
            return {"type": "modify", "rows_affected": cursor.rowcount}
```
